bool Callatz(int n,vector<int> &key,set<int> &num){
    if(find(key.begin(),key.end(),n) == key.end()){
        key.push_back(n);
        while(n != 1){
            if(n%2 == 0){
                n = n/2;
            }else{
                n = (n*3+1)/2;
            }
            if(find(key.begin(),key.end(),n) != key.end()){
                if(num.find(n) != num.end()){
                    num.erase(n);
                }
                return true;
            }else{
                key.push_back(n);
            }
        }
        return true;
    }else{
        return false;
    }
}
