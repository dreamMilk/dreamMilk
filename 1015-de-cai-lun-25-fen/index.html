#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

struct stu{
    int sno,d,c;
};

bool comp(struct stu a,struct stu b){
    if(a.c+a.d != b.c+b.d) return a.c+a.d > b.c+b.d;
    else if(a.d != b.d) return a.d > b.d;
    else return a.sno < b.sno;
}

int main() {
    int n,l,h,cnt;
    vector<stu> s[4];
    scanf("%d %d %d",&n,&l,&h);
    cnt = n;
    for(int i=0;i<n;i++){
        stu j;
        cin >> j.sno >> j.d >> j.c;
        if(j.c < l || j.d < l){
            cnt--;
        }
        else if(j.c >= h && j.d >= h){
            s[0].push_back(j);
        }
        else if(j.d >= h){
            s[1].push_back(j);
        }else if(j.d >= j.c){
            s[2].push_back(j);
        }else{
            s[3].push_back(j);
        }
    }
    printf("%d\n",cnt);
    for(int i=0;i<4;i++){
        sort(s[i].begin(),s[i].end(),comp);
        for(int j=0;j<s[i].size();j++){
            printf("%d %d %d\n",s[i][j].sno,s[i][j].d,s[i][j].c);
        }
    }
    return 0;
}
