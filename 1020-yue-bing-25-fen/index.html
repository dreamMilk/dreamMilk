#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;

struct cake {
    float stock,value,price;
};
bool cmp(cake a,cake b){
    return a.price>b.price;
}

int main() {
    int n,d;
    cin >> n >> d;
    vector<cake> v(n);
    for(int i=0; i<n; i++) {
        cin >> v[i].stock;
    }
    for(int i=0; i<n; i++) {
        cin >> v[i].value;
        v[i].price = v[i].value/v[i].stock;
    }
    sort(v.begin(),v.end(),cmp);
    float account=0;
    for(int i=0;i<n;i++){
        if(d >= v[i].stock){
            account += v[i].value;
            d -= v[i].stock;
        }else{
            account += v[i].price*d;
            break;
        }
    }
    printf("%.2f",account);
    return 0;
}
