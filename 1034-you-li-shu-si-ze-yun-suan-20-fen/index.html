#include<iostream>
#include<cmath>

using namespace std;

int divisor(long long a,long long b){
    return a%b==0?b:divisor(b,a%b);
}

int conversion(long long a,long long b){
    if(b==0){
        cout << "Inf";
        return 0;
    }
    bool flag=((a<0&&b>0) || (a>0&&b<0));
    a = abs(a);b = abs(b);
    long long d=divisor(a,b);
    a /= d;b /= d;
    if(flag) cout << "(-";
    if(a%b==0) cout << a/b;
    else if(a>b) cout << a/b << ' ' << a-a/b*b << '/' << b;
    else cout << a << '/' << b;
    if(flag) cout << ')';
    return 0;
}

int main() {
    long long a1,b1,a2,b2;
    scanf("%lld/%lld %lld/%lld", &a1, &b1, &a2, &b2);
    conversion(a1,b1); cout << " + "; conversion(a2,b2); cout << " = "; conversion(a1*b2+a2*b1,b1*b2); cout << endl;
    conversion(a1,b1); cout << " - "; conversion(a2,b2); cout << " = "; conversion(a1*b2-a2*b1,b1*b2); cout << endl;
    conversion(a1,b1); cout << " * "; conversion(a2,b2); cout << " = "; conversion(a1*a2,b1*b2); cout << endl;
    conversion(a1,b1); cout << " / "; conversion(a2,b2); cout << " = "; conversion(a1*b2,b1*a2); cout << endl;
    return 0;
}
