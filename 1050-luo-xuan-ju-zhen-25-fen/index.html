#include<iostream>
#include<vector>
#include<cmath>
#include<algorithm>

using namespace std;

int cmp(int a,int b){ return a>b; }

int main() {
    int N,m,n;
    cin >> N;
    for(int i=sqrt((double)N);i>0;i--){
        if(N%i == 0) {n=i; m=N/n; break;}
    }
    vector<int> v(N);
    for(int i=0;i<N;i++) scanf("%d",&v[i]);
    sort(v.begin(),v.end(),cmp);
    vector<vector<int>> V(m,vector<int>(n));
    int t=0,level = m%2+m/2;
    for(int i=0;i<level;i++){
        for(int j=i;j<n-i && t < N;j++) V[i][j] = v[t++];
        for(int j=i+1;j<m-i && t < N;j++) V[j][n-i-1] = v[t++];
        for(int j=i+1;j<n-i && t < N;j++) V[m-1-i][n-1-j] = v[t++];
        for(int j=i+1;j<m-1-i && t < N;j++) V[m-1-j][i] = v[t++];
    }
    for(int i=0;i<m;i++){
        for(int j=0;j<n;j++){
            if(j==0) cout << V[i][j];
            else cout << ' ' << V[i][j];
        }
        cout << endl;
    }
    return 0;
}
