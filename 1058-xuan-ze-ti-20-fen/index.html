#include <cstdio>
#include <vector>
#include <set>

using namespace std;

int main() {
    int m,n,cnt;
    char c;
    scanf("%d%d",&n,&m);
    vector<int> score(m),error(m);
    vector<set<char>> options(m);
    for(int i=0;i<m;i++){
        scanf("\n");
        scanf("%d%*d%d",&score[i],&cnt);
        for(int j=0;j<cnt;j++){
            scanf(" %c",&c);
            options[i].insert(c);
        }
    }
    for(int i=0;i<n;i++){
        int num = 0;
        scanf("\n");
        for(int j=0;j<m;j++){
            if(j!=0) scanf(" ");
            set<char> ans;
            scanf("(%d",&cnt);
            for(int k=0;k<cnt;k++){
                scanf(" %c",&c);
                ans.insert(c);
            }
            scanf(")");
            if(ans == options[j]){
                num += score[j];
            }else{
                error[j] ++;
            }
        }
        printf("%d\n",num);
    }
    int max_error = 0;
    for(int i=0;i<m;i++){
        if(error[i] > max_error) max_error=error[i];
    }
    if(max_error==0) printf("Too simple");
    else{
        printf("%d",max_error);
        for(int i=0;i<m;i++){
            if(error[i] == max_error) printf(" %d",i+1);
        }
    }
    return 0;
}
