using namespace std;

struct Node{
    int data,next;
};

int main() {
    int start,n,k;
    Node sqlist[100000];
    scanf("%d %d %d",&start,&n,&k);
    for(int i=0;i<n;i++){
        int address;
        scanf("%d ",&address);
        scanf("%d %d",&sqlist[address].data,&sqlist[address].next);
    }
    int p=start;
    vector<int> v[3];
    while(p!=-1){
        if(sqlist[p].data < 0) v[0].push_back(p);
        else if(sqlist[p].data > k) v[2].push_back(p);
        else v[1].push_back(p);
        p = sqlist[p].next;
    }
    bool flag=false;
    for(int i=0;i<3;i++){
        for(int j=0;j<v[i].size();j++){
            if(flag){
                printf(" %05d\n%05d %d",v[i][j],v[i][j],sqlist[v[i][j]].data);
            }
            else{
                printf("%05d %d",v[i][j],sqlist[v[i][j]].data);
                flag = true;
            }
        }
    }
    printf(" -1");
    return 0;
}
